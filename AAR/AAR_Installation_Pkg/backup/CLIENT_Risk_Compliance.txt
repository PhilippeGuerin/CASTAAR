SELECT c1.snapshot, c1.appname, SUM(c1.checks) AS checks
, SUM(c1.success) AS success

FROM (SELECT c.snapshot, c.appname, c.bc_name
, SUM(c.checks) AS checks, SUM(c.success) AS success

FROM `dfs`.`CLIENT Views`.`CLIENT_QualityModel_Compliance` AS c
WHERE c.bc_name IN ('Robustness', 'Security', 'Efficiency', 'Performance')GROUP BY c.snapshot, c.appname, c.bc_name) AS c1

GROUP BY c1.snapshot, c1.appname
